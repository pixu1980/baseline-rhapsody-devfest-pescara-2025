Write direction agnostic CSS for global reach

        
/* Physical properties (old way) */
.box-physical {
  margin-left: 1rem;
  margin-right: 2rem;
  border-top: 2px solid blue;
  border-bottom: 1px solid gray;
}

/* Logical properties (modern way) */
.box-logical {
  /* Inline = horizontal in LTR/RTL */
  margin-inline-start: 1rem;
  margin-inline-end: 2rem;
  
  /* Block = vertical */
  border-block-start: 2px solid blue;
  border-block-end: 1px solid gray;
}

/* Shorthand */
.element {
  padding-inline: 2rem;  /* left & right */
  padding-block: 1rem;   /* top & bottom */
  margin-inline: auto;   /* center horizontally */
}

/* Size properties */
.card {
  inline-size: 300px;    /* width */
  block-size: 200px;     /* height */
  max-inline-size: 100%; /* max-width */
}
