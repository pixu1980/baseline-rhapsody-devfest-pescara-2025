/* attr(): Access HTML attributes in CSS */
[data-tooltip] {
  position: relative;
}

[data-tooltip]::after {
  /* Read data attribute value */
  content: attr(data-tooltip);
  position: absolute;
  top: 100%;
  left: 50%;
  transform: translateX(-50%);
  background: black;
  color: white;
  padding: 0.5rem;
  border-radius: 4px;
  opacity: 0;
  pointer-events: none;
}

[data-tooltip]:hover::after {
  opacity: 1;
}

/* counter(): Custom list numbering */
ol {
  list-style: none;
  counter-reset: my-counter;
}

li {
  counter-increment: my-counter;
}

li::before {
  content: counter(my-counter) ". ";
  font-weight: bold;
  color: #8b5cf6;
}

/* counters(): Nested counters */
li::before {
  content: counters(section, ".") " ";
}

/* url(): Load external resources */
.element {
  /* Images */
  background-image: url('/path/to/image.jpg');
  
  /* Fonts */
  src: url('/fonts/custom.woff2') format('woff2');
  
  /* SVG as data URI */
  mask-image: url('data:image/svg+xml,...');
}
