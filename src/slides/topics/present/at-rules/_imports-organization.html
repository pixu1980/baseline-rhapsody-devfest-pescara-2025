Import Flow
1. Import into layers ↓
2. Control cascade order ↓
3. Modular & maintainable ✓
Benefits
✓ Organized codebase
✓ Predictable cascade
✓ Easy maintenance
 
        
/* Main stylesheet */
/* Define layer order first */
@layer reset, tokens, base, components, utilities;

/* Import files into layers */
@import url('reset.css') layer(reset);
@import url('tokens.css') layer(tokens);
@import url('base.css') layer(base);

/* Import with supports */
@import url('modern.css') 
  supports(display: grid);

/* Import with media query */
@import url('print.css') 
  print;

/* Combine conditions */
@import url('advanced.css')
  layer(components)
  supports(container-type: inline-size)
  (min-width: 768px);

/* Component imports */
@import url('components/button.css') 
  layer(components);
@import url('components/card.css') 
  layer(components);

/* Utilities last */
@import url('utilities.css') 
  layer(utilities);

/* Modern modular CSS architecture:
   - Organized by concern
   - Controlled cascade with layers  
   - Progressive enhancement
   - Easy to maintain */
