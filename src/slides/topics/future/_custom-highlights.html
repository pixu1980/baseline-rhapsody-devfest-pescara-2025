Different highlight types 
        
/* Define custom highlight */
::highlight(search-results) {
  background-color: yellow;
  color: black;
}

::highlight(spelling-error) {
  text-decoration: wavy underline red;
}

::highlight(grammar-error) {
  text-decoration: wavy underline blue;
}

/* Syntax highlighting */
::highlight(keyword) {
  color: #c678dd;
  font-weight: bold;
}

::highlight(string) {
  color: #98c379;
}

::highlight(comment) {
  color: #5c6370;
  font-style: italic;
}

/* Selection alternatives */
::highlight(user-selection) {
  background: rgba(102, 126, 234, 0.3);
}

/* Inherit and override */
::highlight(important) {
  background: #fef3c7;
  font-weight: bold;
  border-bottom: 2px solid #f59e0b;
}
